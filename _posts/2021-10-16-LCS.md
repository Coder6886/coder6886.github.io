## 题目：

[题目传送门](https://www.acwing.com/problem/content/899/)

## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
int f[1005][1005];
int main(){
    int as,bs;
    cin >> as >> bs;
    string a, b;
    cin >> a >> b;
    a = 'a' + a;
    b = 'b' + b;
    for(int i = 1; i <= as; i++){
        for(int j = 1; j <= bs; j++){
            f[i][j] = max(f[i-1][j],f[i][j-1]);
            if(a[i] == b[j])
                f[i][j] = max(f[i][j],f[i-1][j-1]+1);
        }
    }
    cout << f[as][bs] << endl;
    return 0;
}
```
