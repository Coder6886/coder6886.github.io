# 文件压缩

## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
char a[10005];
char b[10005];
char ans[10005];
bool used[10005];
int ffind(char *beg, char *en, char n){
    char *tmp;
    for(tmp = en-1;tmp >= beg; tmp--){
        if(*tmp == n && !used[tmp-beg]){used[tmp-beg] = 1;return tmp-beg;}
    }
    return -1;
}
int main(){
    int n;
    cin >> n;
    char tmp;
    for(int i = 1; i <= n; i++){
        scanf("%c", &tmp);
        if(isalpha(tmp))
            b[i] = a[i] = tmp ;
        else
            i--;
    }
    int q;
    cin >> q;
    sort(&b[1],&b[n+1]);
    int cnt = 0;
    for(int i = 1; i <= n; i++){
        if(b[i] == a[q]){q = i;break;}
    }
    ans[cnt++] = a[q];
    used[q-1] = 1;
    for(int i = ffind(&b[1],&b[n+1],a[q])+1; cnt < n; i = ffind(&b[1],&b[n+1],a[i])+1){
        ans[cnt++] = a[i];
    }
    while(cnt){
        cout << ans[--cnt];
    }
    return 0;
}
```
