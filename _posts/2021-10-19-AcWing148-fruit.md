## 题目：

[题目传送门](https://www.acwing.com/problem/content/150/)

## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
priority_queue<int,vector<int>,greater<int> > pq;
int main(){
    int n;
    int res  = 0;
    cin >> n;
    for(int i = 0; i < n; i++){
        int tmp;
        cin >> tmp;
        pq.push(tmp);
    }
    while(pq.size() != 1){
        int a = pq.top();pq.pop();
        int b = pq.top();pq.pop();
        res += a + b;
        pq.push(a+b);
    }
    cout << res << endl;
    return 0;
}
```
