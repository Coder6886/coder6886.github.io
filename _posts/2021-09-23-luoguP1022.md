# 计算器的改良

## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
char in[10000];
char getx(){
    for(int i = 0; i < strlen(in); i++){
        if(isalpha(in[i]))return in[i];
    }
}
int main(){
    scanf("%s",in);
    int len = strlen(in);
    long long a = 0,b = 0;//ax=b
    int eql = 1;//equal left
    char x = getx();
    for(int i = 0; i < len;){
        bool neg = false;
        if(in[i] == '=')eql = -1,i++;
        if(in[i] == '-')neg = true,i++;
        if(in[i] == '+')neg = false,i++;
        long long tmp = 0;
        bool changed = false;
        while(isdigit(in[i])&&i<len){
            changed = true;
            tmp = tmp*10 + in[i]-'0';
            i++;
        }
        if(changed == false)tmp = 1;
        if(neg)tmp *= -1;
        if(in[i] == x)a += eql*tmp,i++;
        else b += -eql*tmp;
    }
    double ans = b*1.0/a;
    if(abs(ans)<0.000001)ans = 0.0;
    printf("%c=%.3f\n",x,ans);
    return 0;
}

```
