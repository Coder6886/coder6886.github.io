# 用堆实现堆排序

## 题目：

输入一个长度为 n 的整数数列，从小到大输出前 m 小的数。

## 程序：

## 堆：

```c++
#include<iostream>

using namespace std;
class myheap{//我的堆
    public:
        int h[100010];
        int size;
        //true 是小根堆
        //false 是大根堆
        bool heaptype;
        myheap(){
            size=0;
            heaptype = true;
        }
        bool comp(int a, int b){
            return heaptype?a<b:a>b;
        }
    private:
        void up(int k){
            int x = k;
            while(1){
                int tmp = x;
                if(tmp/2>=1 && !comp(h[tmp/2],h[x]))x=x/2;
                if(x == tmp)return;
                swap(h[tmp],h[x]);
            }
        }
        void down(int k){
            int x = k;
            while(1){
                int tmp= x;
                if(tmp*2<=size && comp(h[2*tmp],h[x]))x=tmp*2;
                if(tmp*2+1<=size && comp(h[2*tmp+1],h[x]))x=tmp*2+1;
                if(x == tmp)return;
                swap(h[tmp],h[x]);
            }
        }
    public:
        void push(int k){
            h[++size] = k;
            up(size);
        }
        int pop(){
            int ret = h[1];
            h[1] = h[size--];
            down(1);
            return ret;
        }
        int top(){
            return h[1];
        }
        int popnth(int n){
            int ret = h[n];
            h[n] = h[size--];
            down(n);
            up(n);
            return ret;
        }
        int getnth(int n){
            return h[n];
        }
        void rewrite(int n, int m){
            h[n] = m;
        }
        int getsize(){
            return size;
        }
};
```

## 主函数

```c++
myheap he;
int main(){
    int n,m;
    cin >> n >> m;
    he.heaptype = true;
    for(int i = 0; i < n; i++){
        int tmp;
        cin >> tmp;
        he.myheap::push(tmp);
    }
    for(int i = 0; i < m; i++){
        cout << he.myheap::pop() << ' ';
    }
    return 0;
}
```
