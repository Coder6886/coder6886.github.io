# 复赛真题

## 1 数字游戏

```c++
#include<bits/stdc++.h>

using namespace std;
int main(){
    int ans = 0;
    for(int i = 0; i < 8; i++){
        char tmp;
        scanf("%c",&tmp);
        if(tmp == '1')ans++;
    }
    cout << ans << endl;
    return 0;
}

```

## 2 公交换乘

```c++
#include<bits/stdc++.h>

using namespace std;
const int maxn=1e5+10;
int history[maxn][3],tickets[maxn][3];
int ticketsh = 0;
int main(){
    int n;
    int ans=0;
    cin >> n;
    for(int i = 0; i< n; i++){
        cin >>history[i][0] >> history[i][1] >> history[i][2];
        if(history[i][0] == 0){
            tickets[ticketsh][0] = history[i][1];
            tickets[ticketsh][1] = history[i][2];
            tickets[ticketsh][2] = 1;
            ticketsh ++;
            ans += history[i][1];
        }
        else{
            int uset= ticketsh;
            for(int j = ticketsh-1;history[i][2]-tickets[j][1]<=45&&j>=0;j--){
                if(tickets[j][0] >= history[i][1] && tickets[j][2] == 1){
                    uset=j;
                }
            }
            if(uset == ticketsh)ans += history[i][1];
            else tickets[uset][2] = 0;
        }
    }
    cout << ans << endl;
    return 0;
}

```
