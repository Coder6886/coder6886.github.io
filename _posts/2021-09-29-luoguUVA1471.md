# UVA1471 Defence Lines
## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
struct tri {
    int first;//a[i]
    int second;//g(i)
    int third;//f(i)
    tri() {
        first = 0;
        second = 0;
        third = 0;
    }
    void settri(int f, int s, int t) {
        first = f;
        second = s;
        third = t;
    }
    void printtri() {
        cout << first << " " << second << " " << third << endl;
    }
    bool operator < (const tri& b) const {
        return first < b.first;
    }
};
int main() {
    int t;
    cin >> t;
    while (t--) {
        set<tri> s;
        vector<tri> v;
        int n;
        cin >> n;
        tri tmp1;
        tmp1.settri(0, 1, 1);
        for (int i = 0; i < n; i++) {
            cin >> tmp1.first;
            if (i && tmp1.first > v[i - 1].first)tmp1.second = v[i - 1].second + 1;
            else tmp1.second = 1;
            v.push_back(tmp1);
        }
        for (int i = n - 2; i >= 0; i--) {
            if (v[i].first < v[i + 1].first)v[i].third = v[i + 1].third + 1;
            else v[i].third = 1;
        }
        int ans = 1;
        s.insert(v[0]);
        for (int i = 1; i < n; i++) {
            set<tri>::iterator it = s.lower_bound(v[i]);
            if (it != s.begin()) {
                tri tmp = *(--it);
                ans = max(ans, v[i].third + tmp.second);
                if (v[i].second <= tmp.second)continue;
            }
            s.erase(v[i]);
            s.insert(v[i]);
            it = s.find(v[i]);
            it++;
            while (it != s.end() && it->first > v[i].first && it->second <= v[i].second)s.erase(it++);
        }
        cout << ans << endl;
    }
    return 0;
}
```
