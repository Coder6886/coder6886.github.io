# 复赛真题

## 1 分糖果

```c++
#include<bits/stdc++.h>

using namespace std;
int main(){
    int n,l,r;
    cin >> n >> l >> r;
    if(r - l > n)cout << n-1 << endl;
    else{
        if(r / n * n - 1 <= r && r / n * n - 1 >= l)cout <<n-1 <<endl;
        else if(l / n * n - 1 + n >= l && l / n * n - 1 + n <= r)cout<< n-1  <<endl;
        else cout << r%n<< endl;
    }
    return 0;
}
```

## 2 插入排序

```c++
#include<bits/stdc++.h>

using namespace std;
pair<int,int> a[8010];
int b[8010];
int main(){
    int n,q;
    scanf("%d%d",&n,&q);
    for(int i = 1; i <= n; i++){
        scanf("%d",&a[i].first);
        a[i].second = i;
    }
    sort(a+1,a+n+1);
    for(int i = 1; i <= n; i++){
        b[a[i].second] = i;
    }
    while(q--){
        int op;
        scanf("%d",&op);
        if(op == 1){
            int x, v;
            scanf("%d%d", &x, &v);
            a[b[x]].first = v;
            int pos = b[x];
            while(pos+1<=n && a[pos] > a[pos+1]){
                swap(a[pos],a[pos+1]);
                b[a[pos].second] = pos;
                b[a[pos+1].second] = pos+1;
                pos++;
            }
            while(pos-1>0 && a[pos] < a[pos-1]){
                swap(a[pos],a[pos-1]);
                b[a[pos].second] = pos;
                b[a[pos-1].second] = pos-1;
                pos--;
            }
        }
        else{
            int x;
            scanf("%d",&x);
            printf("%d\n",b[x]);
        }
    }
    return 0;
}
```

## 3 网络连接

```c++
#include<bits/stdc++.h>

using namespace std;
unordered_map<string,int> um;
bool chk(string adin){
    char in[30];
    for(int i = 0;i < adin.size(); i++){
        in[i] = adin[i];
    }
    in[adin.size()]='\0';
    char tmp[30];
    int a = -1,b = -1,c = -1,d = -1,e = -1;
    const int okabcd = 1 << 8;
    const int oke = 1 << 16;
    if(sscanf(in,"%d.%d.%d.%d:%d",&a,&b,&c,&d,&e) != 5)return false;
    if(a < okabcd && b < okabcd && c < okabcd && d < okabcd && e < oke && a >= 0 && b >= 0 && c >= 0 && d >= 0 && e >= 0){
        sprintf(tmp,"%d.%d.%d.%d:%d",a,b,c,d,e);
        return strcmp(in,tmp) == 0;
    }
    return false;
}
int main(){
    int n;
    cin >> n;
    for(int i = 0; i < n; i++){
        string op, ad;
        cin >> op >> ad;
        if(op == "Client"){
            if(chk(ad)){
                if(um.count(ad)){
                    cout << um[ad] << endl;
                }
                else{
                    cout <<"FAIL"  << endl;
                }
            }
            else{
                cout << "ERR" << endl;
            }
        }
        else{
            if(chk(ad)){
                if(um.count(ad)){
                    cout <<"FAIL" << endl;
                }
                else{
                    um[ad] = i+1;
                    cout << "OK" << endl;
                }
            }
            else{
                cout <<"ERR" << endl;
            }
        }
    }
    return 0;
}
```
