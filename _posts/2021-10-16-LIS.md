## 题目：

[题目传送门](https://www.acwing.com/problem/content/897/)

## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
int f[1005][2];
int main(){
    int n, res = 0;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> f[i][0];
        f[i][1] = 1;
        for(int j = 0; j < i; j++)
            if(f[j][0] < f[i][0] && f[j][1]+1 > f[i][1])
                f[i][1] = f[j][1]+1;
        res = max(res,f[i][1]);
    }
    cout << res << endl;
    return 0;
}
```
