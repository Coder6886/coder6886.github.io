## 1 优秀的拆分

```c++
#include<iostream>

#include<cmath>

using namespace std;
int main(){
    freopen("power.in","r",stdin);
    freopen("power.out","w",stdout);
    int n;
    cin >> n;
    if(n%2){cout << -1 << endl; return 0;}
    while(n){
        cout << (int)pow(2,floor(log2(n))) << " ";
        n = n - (int)pow(2,floor(log2(n)));
    }
    cout << endl;
    return 0;
}
```

## 2 直播获奖





```c++
#include<iostream>

#include<queue>

#include<cmath>

using namespace std;
priority_queue<int> a;//没有获奖的人
priority_queue<int,vector<int>,greater<int> > b;//已经获奖的人
int main(){
    freopen("live.in","r",stdin);
    freopen("live.out","w",stdout);
    int n,w;
    cin >> n >> w;
    for(int i = 1; i <= n; i++){
        int tmp;
        cin >> tmp;
        b.push(tmp);
        while(b.size()&&a.size()&&b.top()<a.top()){
            int x = a.top();
            a.pop();
            a.push(b.top());
            b.pop();
            b.push(x);
        }
        while(b.size()>max(1,i*w/100)){
            a.push(b.top());
            b.pop();
        }
        cout << b.top() << " ";
    }
    cout << endl;
    return 0;
}
```

## 3 表达式

```C++
#include<iostream>
#include<cstring>
#include<string>
#include<stack>
using namespace std;
int n,val[1000010];
char s[1000010];
bool willchange[1000010];
stack<int>stak;
struct node{
    int val=0;
    char t=0xff;
    int idx=-1;
    //int father;
    int lchild=-1,rchild=-1;
    bool willchangeroot=false;
}tree[1000010];
int topp = 0;
void dfs(int x){
    tree[x].willchangeroot = true;
    if(tree[x].t == 'x')return;
    if(tree[x].t == '!')dfs(tree[x].lchild);
    else if(tree[x].t == '&'){
        if(tree[tree[x].lchild].val == 1)dfs(tree[x].rchild);
        if(tree[tree[x].rchild].val == 1)dfs(tree[x].lchild);
    }
    else if(tree[x].t == '|'){
        if(tree[tree[x].lchild].val == 0)dfs(tree[x].rchild);
        if(tree[tree[x].rchild].val == 0)dfs(tree[x].lchild);
    }
}
int main(){
    cin.getline(s,1000010);
    cin >> n;
    for(int i = 1; i <= n; i++){
        scanf("%d",&val[i]);
    }
    int strsize=strlen(s);
    for(int i = 0; i < strsize; i++){
        if(s[i] == 'x'){
            i++;
            int tmp = 0;
            while(s[i] != ' '){
                tmp = tmp * 10 + s[i] - '0';
                i++;
            }
            tree[topp++].t = 'x';
            tree[topp-1].val = val[tmp];
            stak.push(topp-1);
            tree[topp-1].idx = tmp;
        }
        else if(s[i] == '!'){
            i++;
            tree[topp++].t = '!';
            //tree[topp-2].father = topp-1;
            tree[topp-1].lchild = topp-2;
            tree[topp-1].val = !tree[topp-2].val;
            stak.pop();
            stak.push(topp-1);
        }
        else if(s[i] == '&'){
            i++;
            tree[topp++].t = '&';
            int tmp1=stak.top();stak.pop();
            int tmp2=stak.top();stak.pop();
            //tree[tmp1].father = topp-1;
            //tree[tmp2].father = topp-1;
            tree[topp-1].lchild = tmp2;
            tree[topp-1].rchild = tmp1;
            stak.push(topp-1);
            tree[topp-1].val = tree[tmp1].val & tree[tmp2].val;
        }
        else if(s[i] == '|'){
            i++;
            tree[topp++].t = '|';
            int tmp1=stak.top();stak.pop();
            int tmp2=stak.top();stak.pop();
            //tree[tmp1].father = topp-1;
            //tree[tmp2].father = topp-1;
            tree[topp-1].lchild = tmp2;
            tree[topp-1].rchild = tmp1;
            stak.push(topp-1);
            tree[topp-1].val = tree[tmp1].val | tree[tmp2].val;
        }
    }

    dfs(topp-1);
    for(int i = 0; i < topp; i++){
        if(tree[i].t == 'x'){
            willchange[tree[i].idx] = tree[i].willchangeroot;
        }
    }
    int q;
    cin >> q;
    int res = tree[stak.top()].val;
    while(q--){
        int tmp;
        scanf("%d", &tmp);
        if(willchange[tmp])printf("%d\n",!res);
        else printf("%d\n",res);
    }
    return 0;
}
```

```c++
#include<cstring>

#include<string>

using namespace std;
int a[1005][1005];
int dp[1005][1005];
int ans2=-0x3f3f3f3f;
const int direcx[3] = {1,-1,0};
const int direcy[3] = {0,0,1};
int n,m;
int main(){
    #ifdef LOCAL
    
    freopen("infile.in","r",stdin);
    
    #else
    
    freopen("number.in","r",stdin);
    freopen("number.out","w",stdout);
    
    #endif // LOCAL
    
    cin >> n >> m;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < m; j++){
            cin >> a[i][j];dp[i][j] = -0x3f3f3f3f;
        }
    }
    int h = 0;
    for(int i = n-1; i >= 0; i--){
        h += a[i][m-1];
        dp[i][m-1] = h;
    }
    for(int j = m-2; j >= 0; j--){
        for(int i = n-1; i >= 0; i--){
            h = 0;
            for(int k = i; k < n; k++){
                h += a[k][j];
                if(j+1<m)
                    dp[i][j] = max(dp[i][j],dp[k][j+1]+h);
            }
            h = 0;
            for(int k = i; k >= 0; k--){
                h += a[k][j];
                if(j+1<m)
                    dp[i][j] = max(dp[i][j],dp[k][j+1]+h);
            }
        }
    }
    cout << dp[0][0] << endl;
    return 0;
}
```
