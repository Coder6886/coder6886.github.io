# dijkstra

# 849

## 题目

[题目传送门](https://www.acwing.com/problem/content/851/)

## 代码

```c++
#include<bits/stdc++.h>

using namespace std;
int e[505][500];
int dist[505];
bool s[505];
int dijkstra(int start, int finish, int n){
    memset(s,false,sizeof s);
    memset(dist,0x3f,sizeof dist);
    dist[start] = 0;
    while(1){
        int t = -1;
        for(int i = 1; i <= n; i++){
            if(!s[i] && (t == -1 || dist[i] < dist[t]))t = i;
        }
        if(t == -1)break;
        s[t] = true;
        int k = dist[t];
        for(int i = 1; i <= n; i++){
            dist[i] = min(k+e[t][i],dist[i]);
        }
    }
    if(dist[finish] == 0x3f3f3f3f)return -1;
    return dist[finish];
}
int main(){
    memset(e,0x3f,sizeof e);
    int n,m;
    cin >> n >> m;
    for(int i = 0; i < m; i++){
        int a,b,c;
        cin >> a >> b >> c;
        e[a][b] = min(e[a][b],c);
    }
    cout << dijkstra(1,n,n) << endl;
    return 0;
}
```

# 850

## 题目

[题目传送门](https://www.acwing.com/problem/content/852/)

## 代码

```c++
#include<bits/stdc++.h>

using namespace std;
const int N = 1e4*15 + 10;
int h[N], e[N], ne[N], w[N], idx;
int dist[N];
bool s[N];
typedef pair<int,int> PII;
void add(int a, int b, int weight){
    e[idx] = b, w[idx] = weight, ne[idx] = h[a], h[a] = idx++;
}
int dijkstra(int start, int finish, int n){
    memset(s,false,sizeof s);
    memset(dist,0x3f,sizeof dist);
    priority_queue<PII,vector<PII>,greater<PII> > mheap;
    mheap.push({0,start});
    dist[start] = 0;
    while(mheap.size()){
        PII t = mheap.top();
        mheap.pop();
        int point = t.second, len = t.first;
        if(s[point])continue;
        s[point] = true;
        for(int i = h[point]; i != -1; i = ne[i]){
            int j = e[i];
            if(dist[j] > len+w[i]){
                dist[j] = len+w[i];
                mheap.push({dist[j],j});
            }
        }
    }
    if(dist[finish] == 0x3f3f3f3f)return -1;
    return dist[finish];
}
int main(){
    memset(h,-1,sizeof h);
    int n,m;
    cin >> n >> m;
    for(int i = 0; i < m; i++){
        int a,b,c;
        cin >> a >> b >> c;
        add(a,b,c);
    }
    cout << dijkstra(1,n,n) << endl;
    return 0;
}
```
