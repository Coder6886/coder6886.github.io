# 回文数

## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
struct bignum{
    vector<int>val;
    int n;//进制
    bignum(){
        n = 10;
    }

};
void print(bignum a){
    for(int i = a.val.size()-1; i >= 0; i--){
        std::cout << a.val[i];
    }
}
void getin(bignum &a){
    std::cin >> a.n;
    bool f16 = false;
    if(a.n == 16){f16 = true;}
    std::stack<int>tmp;
    char c;
    while(std::cin >> c && (isdigit(c) || f16 == true && (isdigit(c) || c >= 'A' && c <= 'F'))){
        if(isalpha(c))tmp.push(c-'A'+10);
        else tmp.push(c-'0');
    }
    while(tmp.size()){
        a.val.push_back(tmp.top());
        tmp.pop();
    }
}
bignum p(bignum a, bignum b){
    //同样进制同样长度的加法
    int in = 0;
    bignum ans;
    ans.n = a.n;
    for(int i = 0; i < a.val.size(); i++){
        ans.val.push_back((a.val[i]+b.val[i]+in)%a.n);
        in = (a.val[i]+b.val[i]+in)/a.n;
    }
    if(in)ans.val.push_back(1);
    return ans;
}
bool ispalindrome(bignum a){
    for(int i = 0; i < a.val.size()/2; i++){
        if(a.val[i] != a.val[a.val.size()-1-i]){
            return false;
        }
    }
    return true;
}
bignum& rev(bignum &a){
    reverse(a.val.begin(),a.val.end());
    return a;
}

int main(){
    bignum a;
    getin(a);
    int ans = 0;
    while(++ans <= 30 && !ispalindrome(p(rev(a),a))){
        a = p(rev(a),a);
    }
    if(ans <= 30){
        cout << "STEP=" <<ans <<endl;
        return 0;
    }
    cout << "Impossible!" << endl;
    return 0;
}
```
