#  UVA12265 Selling Land
## 代码：

```c++
#include<bits/stdc++.h>

using namespace std;
int a[1005][1005];
int c[4010];
int main() {
	int t;
	int n, m;
	cin >> t;
	while (t--) {
		memset(c, 0, sizeof c);
		memset(a, 0, sizeof a);
		cin >> n >> m;
		vector<int> h;
		char tmp;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				scanf("%c", &tmp);
				if (tmp != '.' && tmp != '#') {
					j--;
				}
				if (tmp == '.') {
					a[i][j] = 1;
				}
			}
		}
		h.resize(0);
		h.resize(m, 0);
		for (int i = 0; i < n; i++) {
			stack<pair<int, int> > height;
			for (int j = 0; j < m; j++) {
				if (a[i][j]) {
					h[j]++;
				}
				else {
					h[j] = 0;
					stack<pair<int, int> >().swap(height);
					continue;
				}
				pair<int, int> p = { j,h[j] };
				while (height.size() && ( height.top().second >= h[j])) {
					p.first = height.top().first;
					height.pop();
				}
				if(!height.size() || height.top().second - height.top().first < h[j] - p.first)
					height.push(p);
				c[2 * (j-height.top().first + 1 + height.top().second)]++;
			}
		}
		for (int i =4; i <= 4000; i++) {
			if (c[i])printf("%d x %d\n", c[i], i);
		}
	}
	
	return 0;
}
```
